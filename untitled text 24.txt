cacheSolve <- function(x, ...) {
 
  inver <- x$getInverse()
  if (!is.null(inver)) {
    message("getting cached data")
    return(inver)
  }
  matri <- x$get()
  inver <- solve(matri, ...)
  x$setInverse(inver)
  inver
}
